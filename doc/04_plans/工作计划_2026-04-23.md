#### Step 3：实现 agent_service

**`backend/services/agent_service.py` 完整实现：**

```python
import json
import httpx
from backend.config import settings

SYSTEM_PROMPT = """你是一位专业的校园空间设计师，擅长将普通人的日常感受翻译为专业的设计改进方案。

用户会用非常朴素的语言描述他们在校园某个位置的感受和期望。
你的任务是：
1. 理解用户真正想要的空间体验（不是字面意思，而是背后的需求）
2. 将这个需求翻译为具体的、可执行的设计说明

输出格式为 JSON，包含两个字段：
- "ai_response": 用温暖的第一人称语气向用户解释你理解了什么（2-3句话，像在和朋友聊天）
- "design_description": 具体的设计改进说明（面向图生图模型的 prompt，用英文描述，包含材质、颜色、空间关系）

示例输入：
  位置：图书馆前庭
  情绪：太单调
  描述：这里的长椅太少了，下课想坐下来发发呆

示例输出：
{
  "ai_response": "你提到希望这里能有更多停留和休息的地方。我注意到这个区域午后阳光充足但缺少遮阴，下课时段人流较大却没有足够的座位。",
  "design_description": "Add a lightweight timber pergola structure on the east side of the library forecourt, with low cedar planters and three groups of wooden benches with backrests. Preserve the existing open sightline to the main entrance. Use warm-toned materials that complement the building facade."
}"""


async def translate_to_design_spec(
    emotion_tags: list[str] | None,
    user_input: str,
    location_label: str | None,
    location_x: float | None = None,
    location_y: float | None = None,
) -> dict:
    """调用 LLM API，将用户自然语言翻译为结构化设计说明"""

    if not settings.LLM_API_KEY or not settings.LLM_API_URL:
        # 未配置 LLM 时返回兜底响应
        return {
            "ai_response": f"你描述了对「{location_label or '这个位置'}」的感受。AI 正在分析你的创意，准备生成设计方案。",
            "design_description": f"Improve the spatial experience at {location_label or 'the selected location'} based on user feedback about {', '.join(emotion_tags or ['general improvement'])}.",
        }

    # 构建用户消息
    parts = []
    if location_label:
        parts.append(f"位置：{location_label}")
    if emotion_tags:
        parts.append(f"情绪/感受：{', '.join(emotion_tags)}")
    parts.append(f"描述：{user_input}")

    user_message = "\n".join(parts)

    # 调用 LLM API（OpenAI 兼容格式，大多数服务商都支持）
    async with httpx.AsyncClient(timeout=60.0) as client:
        response = await client.post(
            settings.LLM_API_URL,
            headers={
                "Authorization": f"Bearer {settings.LLM_API_KEY}",
                "Content-Type": "application/json",
            },
            json={
                "model": settings.LLM_MODEL,
                "messages": [
                    {"role": "system", "content": SYSTEM_PROMPT},
                    {"role": "user", "content": user_message},
                ],
                "temperature": 0.7,
                "max_tokens": 800,
            },
        )
        response.raise_for_status()
        data = response.json()

    # 解析响应
    content = data["choices"][0]["message"]["content"]

    # 尝试从响应中提取 JSON
    try:
        # 处理 markdown code block 包裹的情况
        if "```json" in content:
            content = content.split("```json")[1].split("```")[0].strip()
        elif "```" in content:
            content = content.split("```")[1].split("```")[0].strip()

        result = json.loads(content)
        return {
            "ai_response": result.get("ai_response", ""),
            "design_description": result.get("design_description", ""),
        }
    except (json.JSONDecodeError, KeyError):
        # JSON 解析失败时，整段作为 ai_response
        return {
            "ai_response": content,
            "design_description": "",
        }
```

> **注意：** 如果使用 Claude API（非 OpenAI 兼容格式），需要调整请求体的 `messages` 结构和响应解析。大多数国产大模型（通义、DeepSeek、智谱、百川等）都兼容 OpenAI 格式。

### 2.4 候选方案对比

| 方案 | 优点 | 缺点 | 适用场景 |
|------|------|------|---------|
| Stable Diffusion + ControlNet (自部署) | 可控性强，免费 | 需要 GPU 服务器，部署复杂 | 有 GPU 资源 |
| ComfyUI (自部署) | 灵活的工作流 | 同上 + 学习成本 | 需要精细控制 |

| 通义万相 / 文心一格 | 国产，中文友好 | 风格可控性一般 | 国内网络环境 |
| DALL-E 3 API | OpenAI 生态，简单 | 不支持图生图（仅文生图） | 无参考图时 |

**推荐方案：** 先用文生图 API（DALL-E 3 或通义万相）快速跑通流程，后续再切换为 ControlNet 图生图提升质量。

    const overlay = document.getElementById('modalOverlay');
    const inputText = document.getElementById('modalInputText');

    const modalImage = document.getElementById('modalImage');

        const res = await API.getDesigns({}); // TODO: api.js 需要添加 getDesignById 方法

        const detailData = await detailRes.json();

        if (detailData.code === 0 && detailData.data) {
            const d = detailData.data;
            inputText.textContent = d.user_input || d.location_label || '-';
            resultText.textContent = d.ai_response || d.design_description || '-';

            if (d.generated_image && d.generated_image !== '生成中...') {
                modalImage.src = d.generated_image;
                modalImage.style.display = 'block';
            }
        }
    } catch (err) {
        console.error('获取设计详情失败:', err);
        inputText.textContent = APP_STATE.language === 'zh' ? '加载失败' : 'Failed to load';
        resultText.textContent = '';
    }
}
